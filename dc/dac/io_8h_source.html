381 // collect data items impl
382 void RawObs::collect_data_items() {
383  m_data_items.clear();
384  std::vector<DataItem> data_items{};
385  auto node_data_items = this->config().get_node("data_items");
386  auto n_data_items = node_data_items.size();
387  for (std::size_t i = 0; i < n_data_items; ++i) {
388  SPDLOG_INFO("add data item {} of {}", i, n_data_items);
389  data_items.emplace_back(
390  config_t{node_data_items[i], this->config().filepath()});
391  }
392  m_data_items = std::move(data_items);
393 
394  sort(m_data_items.begin(), m_data_items.end(), [] (const RawObs::DataItem& a, const RawObs::DataItem& b) {
395 
396  if ((a.interface().find("toltec"))!=std::string::npos && (b.interface().find("toltec"))!=std::string::npos) {
397  std::string cmp_a { a.interface().begin() + 6, a.interface().end() };
398  std::string cmp_b { b.interface().begin() + 6, b.interface().end() };
399  return std::stoi(cmp_a) < std::stoi(cmp_b);
400  }
401  else {
402  return false;
403  }
404  });
405 
406  SPDLOG_DEBUG("collected n_data_items={}\n{}", this->n_data_items(),
407  this->data_items());
408  // update the data indices
409  m_kidsdata_indices.clear();
410  m_teldata_index.reset();
411  m_hwpdata_index.reset();
412  std::smatch m;
413  for (std::size_t i = 0; i < m_data_items.size(); ++i) {
414  if (std::regex_match(m_data_items[i].interface(), m,
415  re_interface_kidsdata)) {
416  m_kidsdata_indices.push_back(i);
417  }
418  if (std::regex_match(m_data_items[i].interface(), m,
419  re_interface_teldata)) {
420  if (m_teldata_index.has_value()) {
421  throw std::runtime_error("found too many telescope data items");
422  }
423  m_teldata_index = i;
424  }
425  if (std::regex_match(m_data_items[i].interface(), m,
426  re_interface_hwpdata)) {
427  if (m_hwpdata_index.has_value()) {
428  throw std::runtime_error(
429  "found too many halfwave plate data items");
430  }
431  m_hwpdata_index = i;
432  }
433  }
434  if (!m_teldata_index) {
435  throw std::runtime_error("no telescope data item found");
436  }
437  // The hwp data is optional
438  if (!m_hwpdata_index) {
439  SPDLOG_INFO("no hwp data item found");
440  }
441  SPDLOG_INFO("kidsdata_indices={} teldata_index={} hwpdata_index={}",
442  m_kidsdata_indices, m_teldata_index, m_hwpdata_index);
443  SPDLOG_INFO("kidsdata={} teldata={} hwpdata={}", kidsdata(), teldata(),
444  hwpdata());
445 }
