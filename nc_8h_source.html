345  template <typename T = node_t, typename... Args>
346  bool has_node(key_t key, Args &&... keys) {
347  if constexpr (std::is_same_v<T, node_t>) {
348  for (const auto &k : {key, keys...}) {
349  if (!(has_node<NcVar>(k) || has_node<NcDim>(k) ||
350  has_node<NcGroupAtt>(k))) {
351  return false;
352  }
353  }
354  return true;
355  } else {
356  if constexpr (sizeof...(keys) == 0) {
357  nodes.emplace(std::piecewise_construct, std::make_tuple(key),
358  std::make_tuple());
359  auto getnode = [&]() {
360  if constexpr (std::is_same_v<T, NcVar>) {
361  return ncfile.getVar(_[key]);
362  } else if constexpr (std::is_same_v<T, NcDim>) {
363  return ncfile.getDim(_[key]);
364  } else if constexpr (std::is_same_v<T, NcGroupAtt>) {
365  return ncfile.getAtt(_[key]);
366  }
367  };
368  constexpr auto gettype = []() {
369  if constexpr (std::is_same_v<T, NcVar>) {
370  return "var";
371  } else if constexpr (std::is_same_v<T, NcDim>) {
372  return "dim";
373  } else if constexpr (std::is_same_v<T, NcGroupAtt>) {
374  return "att";
375  }
376  };
377  if (const auto &v = getnode(); !v.isNull()) {
378  SPDLOG_TRACE("{} key={} value={} ", gettype(), key,
379  nc_utils::pprint{v});
380  nodes[key] = v;
381  return true;
382  }
383  SPDLOG_TRACE("{} key={} ({}) not found", gettype(), key,
384  _[key]);
385  nodes.erase(key);
386  return false;
387  } else {
388  for (const auto &k : {key, keys...}) {
389  if (!has_node<T>(k)) {
390  return false;
391  }
392  }
393  return true;
394  }
395  }
396  }
397  template <typename... Args> bool has_var(Args &&... keys) {
398  return has_node<NcVar>(key_t{keys}...);
399  }
400  template <typename... Args> bool has_dim(Args &&... keys) {
401  return has_node<NcDim>(key_t{keys}...);

415  const auto &att(key_t key) { return node<NcGroupAtt>(key); }
416  keymap_t _;
