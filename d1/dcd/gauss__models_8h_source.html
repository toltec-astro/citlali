96  typename std::enable_if<ND == 2, T>::type meshgrid(const InputDataBasisType& x, const InputDataBasisType& y) const {
97  // column major
98  // [x0, y0,] [x1, y0] [x2, y0] ... [xn, y0] [x0, y1] ... [xn, yn]
99  const long nx = x.size(), ny = y.size();
100  InputDataType xy(nx * ny, 2);
101  // map xx [ny, nx] to the first column of xy
102  Eigen::Map<Eigen::MatrixXd> xx(xy.data(), ny, nx);
103  // map yy [ny, nx] to the second column of xy
104  Eigen::Map<Eigen::MatrixXd> yy(xy.data() + xy.rows(), ny, nx);
105  // populate xx such that each row is x
106  for (Eigen::Index i = 0; i < ny; ++i) {
107  xx.row(i) = x.transpose();
108  }
109  // populate yy such that each col is y
110  for (Eigen::Index j = 0; j < nx; ++j) {
111  yy.col(j) = y;
112  }
113  return xy;
114  }

247  template <typename T>
248  bool operator()(const T* const p, T* r) const {
249  auto cost2 = cos(p[5]) * cos(p[5]);
250  auto sint2 = sin(p[5]) * sin(p[5]);
251  auto sin2t = sin(2. * p[5]);
252  auto xstd2 = p[3] * p[3];
253  auto ystd2 = p[4] * p[4];
254  auto a = - 0.5 * ((cost2 / xstd2) + (sint2 / ystd2));
255  auto b = - 0.5 * ((sin2t / xstd2) - (sin2t / ystd2));
256  auto c = - 0.5 * ((sint2 / xstd2) + (cost2 / ystd2));
257 
258  for (int i=0; i < this->values(); ++i){
259  if (this->sigma->coeffRef(i) != 0) {
260  r[i] = (
261  this->ydata->coeffRef(i) -
262  p[0] * exp(
263  pow(this->xdata->coeffRef(i, 0) - p[1], 2) * a +
264  (this->xdata->coeffRef(i, 0) - p[1]) * (this->xdata->coeffRef(i, 1) - p[2]) * b +
265  pow(this->xdata->coeffRef(i, 1) - p[2], 2) * c
266  )
267  ) / this->sigma->coeffRef(i);
268  }
269  // remove zero weights
270  else {
271  r[i] = (
272  this->ydata->coeffRef(i) -
273  p[0] * exp(
274  pow(this->xdata->coeffRef(i, 0) - p[1], 2) * a +
275  (this->xdata->coeffRef(i, 0) - p[1]) * (this->xdata->coeffRef(i, 1) - p[2]) * b +
276  pow(this->xdata->coeffRef(i, 1) - p[2], 2) * c
277  )
278  ) * this->sigma->coeffRef(i);
279  }
280  }
281  return true;
