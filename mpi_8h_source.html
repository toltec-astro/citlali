64 auto call_with_node_ranks(int rank, const mxx::comm &comm, Func &&func) {
65  using index_t = decltype(comm.size());
66  char node_name[MPI_MAX_PROCESSOR_NAME];
67  int node_len;
68  MPI_Get_processor_name(node_name, &node_len);
69  // gather all processor names to rank
70  std::vector<char> node_names_raw =
71  mxx::gatherv(node_name, SIZET(MPI_MAX_PROCESSOR_NAME), rank, comm);
72  if (rank == comm.rank()) {
73  std::vector<std::string> node_names;
74 
75  for (index_t i = 0; i < comm.size(); ++i) {
76  node_names.emplace_back(node_names_raw.data() +
77  i * MPI_MAX_PROCESSOR_NAME,
78  MPI_MAX_PROCESSOR_NAME);
79  }
80  // SPDLOG_TRACE("node names: {}", node_names);
81  std::unordered_map<std::string, std::vector<index_t>> node_ranks_;
82  for (index_t i = 0; i < comm.size(); ++i) {
83  node_ranks_[node_names[SIZET(i)]].push_back(i);
84  }
85  std::vector<std::pair<std::string, std::vector<index_t>>> node_ranks;
86  for (auto &[node, ranks] : node_ranks_) {
87  std::sort(ranks.begin(), ranks.end());
88  node_ranks.emplace_back(node, ranks);
89  }
90  // sort the vector of node names by first rank
91  std::sort(node_ranks.begin(), node_ranks.end(),
92  [](const auto &lhs, const auto &rhs) {
93  return lhs.second.front() < rhs.second.front();
94  });
95  FWD(func)(std::move(node_ranks));
96  }

140  }
141  }
142  }
143  });
144  MPI_Bcast(&this->m_n_nodes, 1, MPI_INT, master_rank, *this);
145  if (rank() != master_rank) {
146  MPI_Recv(&this->m_node_index, 1, MPI_INT, master_rank, tag, *this,
147  MPI_STATUS_IGNORE);

220  static auto allocate_shared(int master_rank, index_t size,
221  const mxx::comm &comm, MPI_Win *p_win) {
222  // here the MPI APIs works with size of type MPI_Aint, so we
223  // need to be careful about overflow issue for large array
224  Span arr(nullptr, size);
225  auto mem_size = meta::size_cast<MPI_Aint>(arr.size * disp_unit);
226  // allocate
227  MPI_Win_allocate_shared(comm.rank() == master_rank ? mem_size : 0,
228  arr.disp_unit, MPI_INFO_NULL, comm, &arr.data,
229  p_win);
230  if (comm.rank() == master_rank) {
231  SPDLOG_TRACE("shared memory allocated {}", arr);
232  } else {
233  // update the pointer to point to the shared memory on the
234  // creation rank
235  MPI_Aint _mem_size;
236  Span::disp_unit_t _disp_unit;
237  MPI_Win_shared_query(*p_win, master_rank, &_mem_size, &_disp_unit,
238  &arr.data);
239  // update size
240  arr.size = meta::size_cast<Span::index_t>(_mem_size) / disp_unit;
241  assert(_disp_unit == arr.disp_unit);
242  SPDLOG_TRACE("shared memroy connected from rank={} {}", comm.rank(),
243  arr);
244  }
245  return arr;
246  }
