794  // fit the maps

797  init_fwhm, init_row, init_col);

800  perrors.row(i) = det_perror;

803  // otherwise keep value from previous iteration
804  else {
805  params.row(i) = p0.row(i);
806  perrors.row(i) = perror0.row(i);
807  }
808 
809  return 0;
810  });
811 
812  logger->info("number of good fits {}/{}", good_fits.cast<double>().sum(), n_maps);
813  }
814  }
815 
816  // increment loop iteration
817  current_iter++;
818 
819  if (current_iter < beammap_iter_max) {
820  // check if all detectors are converged
821  if ((converged.array() == true).all()) {
822  logger->info("all maps converged");
823  keep_going = false;
824  }
825  else if (current_iter > 1) {
826  // only do convergence test if tolerance is above zero, otherwise run all iterations
827  if (beammap_iter_tolerance > 0) {
828  // loop through maps and check if it is converged
829  logger->info("checking convergence");
830  grppi::map(tula::grppi_utils::dyn_ex(omb.parallel_policy), det_in_vec, det_out_vec, [&](auto i) {
831  if (!converged(i)) {
832  // get relative change from last iteration
833  Eigen::ArrayXd diff;
834  if (!ptcproc.run_fruit_loops) {
835  diff = abs((params.row(i).array() - p0.row(i).array())/p0.row(i).array());
836  }
837  else {
838  diff = abs(omb_copy.signal[i].array() - omb.signal[i].array()/omb_copy.signal[i].array());
839  }
840  // if a variable is constant, make sure no nans are present
841  auto d = (diff.array()).isNaN().select(0,diff);
842  if ((d.array() <= beammap_iter_tolerance).all()) {
843  // set as converged
844  converged(i) = true;
845  // set convergence iteration
846  converge_iter(i) = current_iter;
847  }
848  }
849  return 0;
850  });
851 
852  logger->info("{} maps converged on iter {}", (converged.array() == true).count(), current_iter);
853 
854  // stop if all maps converged
855  if ((converged.array() == true).all()) {
856  logger->info("all maps converged");
857  keep_going = false;
858  }
859  }
860  else {
861  logger->info("bypassing convergence check");
862  }
863 
864  }
865 
866  // set previous iteration fits to current iteration fits
867  p0 = params;
868  perror0 = perrors;
869  }
870  else {
871  logger->info("max iteration reached");
872  keep_going = false;
873  }
874  }
