205 template <class calib_t, typename Derived>
206 auto RTCProc::calc_map_indices(calib_t &calib, Eigen::DenseBase<Derived> &det_indices, Eigen::DenseBase<Derived> &nw_indices,
207  Eigen::DenseBase<Derived> &array_indices, Eigen::DenseBase<Derived> &fg_indices,
208  std::string map_grouping) {
209  // indices for maps
210  Eigen::VectorXI indices(array_indices.size()), map_indices(array_indices.size());
211 
212  // number of maps from grouping
213  int n_maps = 0;
214 
215  // overwrite map indices for networks
216  if (map_grouping == "nw") {
217  indices = nw_indices;
218  n_maps = calib.n_nws;
219  }
220  // overwrite map indices for arrays
221  else if (map_grouping == "array") {
222  indices = array_indices;
223  n_maps = calib.n_arrays;
224  }
225  // overwrite map indices for detectors
226  else if (map_grouping == "detector") {
227  indices = det_indices;
228  n_maps = calib.n_dets;
229  }
230  // overwrite map indices for fg
231  else if (map_grouping == "fg") {
232  indices = fg_indices;
233  n_maps = calib.fg.size()*calib.n_arrays;
234  }
235  // start at 0
236  if (map_grouping != "fg") {
237  Eigen::Index map_index = 0;
238  map_indices(0) = 0;
239  // loop through and populate map indices
240  for (Eigen::Index i=0; i<indices.size()-1; ++i) {
241  // if next index is larger than current index, increment map index
242  if (indices(i+1) > indices(i)) {
243  map_index++;
244  }
245  map_indices(i+1) = map_index;
246  }
247  }
248  else {
249  // convert fg to indices
250  std::map<Eigen::Index, Eigen::Index> fg_to_index, array_to_index;
251 
252  // get mapping from fg to map index
253  for (Eigen::Index i=0; i<calib.fg.size(); ++i) {
254  fg_to_index[calib.fg(i)] = i;
255  }
256  // get mapping from fg to map index
257  for (Eigen::Index i=0; i<calib.arrays.size(); ++i) {
258  array_to_index[calib.arrays(i)] = i;
259  }
260  // allocate map indices from fg
261  for (Eigen::Index i=0; i<indices.size(); ++i) {
262  map_indices(i) = fg_to_index[indices(i)] + calib.fg.size()*array_to_index[array_indices(i)];
263  }
264  }
265 
266  // return the map indices
267  return std::move(map_indices);
268 }

329  }

339  if (run_despike) {

475  }
476 
477  // copy scan indices
478  out.scan_indices.data = in.scan_indices.data;

495  in.angle.data.resize(0,0);

498 }
