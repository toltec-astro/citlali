813 template <TCProc::SourceType source_type, TCDataKind tcdata_t, typename Derived, typename apt_t>
814 void TCProc::add_gaussian(TCData<tcdata_t, Eigen::MatrixXd> &in, Eigen::DenseBase<Derived> &params, std::string &pixel_axes,
815  std::string &map_grouping, apt_t &apt, double pixel_size_rad, Eigen::Index n_rows, Eigen::Index n_cols) {
816 
817  Eigen::Index n_dets = in.scans.data.cols();
818  Eigen::Index n_pts = in.scans.data.rows();
819 
820  // loop through detectors
821  for (Eigen::Index i=0; i<n_dets; ++i) {
822  // detector index in apt
823  auto det_index = in.det_indices.data(i);
824  // map index
825  auto map_index = in.map_indices.data(i);
826 
827  double az_off = apt["x_t"](det_index);
828  double el_off = apt["y_t"](det_index);
829 
830  // get pointing
831  auto [lat, lon] = engine_utils::calc_det_pointing(in.tel_data.data, az_off, el_off, pixel_axes,
832  in.pointing_offsets_arcsec.data, map_grouping);
833 
834  // get parameters from current map
835  double amp = params(map_index,0);
836  // rows
837  double off_lat = params(map_index,2);
838  // cols
839  double off_lon = params(map_index,1);
840  // row fwhm
841  double sigma_lat = params(map_index,4);
842  // col fwhm
843  double sigma_lon = params(map_index,3);
844  // rot angle
845  double rot_ang = params(map_index,5);
846 
847  // use maximum of sigmas due to atmospheric cleaning
848  double sigma = std::max(sigma_lat, sigma_lon);
849 
850  // subtract source
851  if constexpr (source_type==NegativeGaussian) {
852  amp = -amp;
853  }
854 
855  // rescale offsets and stddev to on-sky units
856  off_lat = pixel_size_rad*(off_lat - (n_rows)/2);
857  off_lon = pixel_size_rad*(off_lon - (n_cols)/2);
858 
859  // convert to on-sky units
860  sigma_lon = pixel_size_rad*sigma;
861  sigma_lat = pixel_size_rad*sigma;
862  sigma = pixel_size_rad*sigma;
863 
864  // get 2d elliptical gaussian angles
865  auto cost2 = cos(rot_ang) * cos(rot_ang);
866  auto sint2 = sin(rot_ang) * sin(rot_ang);
867  auto sin2t = sin(2. * rot_ang);
868  auto xstd2 = sigma * sigma;
869  auto ystd2 = sigma * sigma;
870  auto a = - 0.5 * ((cost2 / xstd2) + (sint2 / ystd2));
871  auto b = - 0.5 * ((sin2t / xstd2) - (sin2t / ystd2));
872  auto c = - 0.5 * ((sint2 / xstd2) + (cost2 / ystd2));
873 
874  // calculate distance to source to truncate it
875  //auto dist = ((lat.array() - off_lat).pow(2) + (lon.array() - off_lon).pow(2)).sqrt();
876 
877  Eigen::VectorXd gauss(n_pts);
878  // make timestream from 2d gaussian
879  for (Eigen::Index j=0; j<n_pts; ++j) {
880  gauss(j) = amp*exp(pow(lon(j) - off_lon, 2) * a +
881  (lon(j) - off_lon) * (lat(j) - off_lat) * b +
882  pow(lat(j) - off_lat, 2) * c);
883  }
884 
885  // check for bad fit?
886  if (!gauss.array().isNaN().any()) {
887  // add gaussian to detector scan
888  in.scans.data.col(i) = in.scans.data.col(i).array() + gauss.array();
889  }
890  }
891 }
