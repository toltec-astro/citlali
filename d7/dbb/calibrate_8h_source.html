123 auto Calibration::calc_tau(Eigen::DenseBase<Derived> &elev, double tau_225_GHz) {
124 
125  // tau at toltec freqs
126  std::map<int,Eigen::VectorXd> tau_freq;
127 
128  // zenith angle
129  auto cz = cos(pi/2 - elev.derived().array());
130  auto secz = 1. / cz;
131  auto A = secz * (1. - 0.0012 * (pow(secz.array(), 2) - 1.));
132 
133  // observed 225 GHz tau
134  auto obs_tau_i = A*tau_225_GHz;
135 
136  // 225 GHz transmission
137  auto tx = (-obs_tau_i).exp();
138 
139  // a1100
140  auto tx_a1100 = tau_polynomial(tx_ratio_coeff["a1100"],elev).array()*tx.array();
141  tau_freq[0] = -(tx_a1100.array().log())/A.array();
142 
143  // a1400
144  auto tx_a1400 = tau_polynomial(tx_ratio_coeff["a1400"],elev)*tx.array();
145  tau_freq[1] = -(tx_a1400.array().log())/A.array();
146 
147  // a2000
148  auto tx_a2000 = tau_polynomial(tx_ratio_coeff["a2000"],elev)*tx.array();
149  tau_freq[2] = -(tx_a2000.array().log())/A.array();
150 
151  return tau_freq;
