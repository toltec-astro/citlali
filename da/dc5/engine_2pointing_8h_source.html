392 void Pointing::fit_maps() {
393  // fit maps
394  logger->info("fitting maps");
395  // placeholder vectors for grppi map
396  std::vector<int> map_in_vec, map_out_vec;
397 
398  map_in_vec.resize(n_maps);
399  std::iota(map_in_vec.begin(), map_in_vec.end(), 0);
400  map_out_vec.resize(n_maps);
401 
402  double init_row = -99;
403  double init_col = -99;
404 
405  // loop through maps
406  grppi::map(tula::grppi_utils::dyn_ex(parallel_policy), map_in_vec, map_out_vec, [&](auto i) {
407  auto array = maps_to_arrays(i);
408  // init fwhm in pixels
409  double init_fwhm = toltec_io.array_fwhm_arcsec[array]*ASEC_TO_RAD/omb.pixel_size_rad;
410  auto [map_params, map_perror, good_fit] =
411  map_fitter.fit_to_gaussian<engine_utils::mapFitter::pointing>(omb.signal[i], omb.weight[i], init_fwhm, init_row, init_col);
412  params.row(i) = map_params;
413  perrors.row(i) = map_perror;
414 
415  if (good_fit) {
416  // rescale fit params from pixel to on-sky units
417  params(i,1) = RAD_TO_ASEC*omb.pixel_size_rad*(params(i,1) - (omb.n_cols)/2);
418  params(i,2) = RAD_TO_ASEC*omb.pixel_size_rad*(params(i,2) - (omb.n_rows)/2);
419  params(i,3) = RAD_TO_ASEC*STD_TO_FWHM*omb.pixel_size_rad*(params(i,3));
420  params(i,4) = RAD_TO_ASEC*STD_TO_FWHM*omb.pixel_size_rad*(params(i,4));
421 
422  // rescale fit errors from pixel to on-sky units
423  perrors(i,1) = RAD_TO_ASEC*omb.pixel_size_rad*(perrors(i,1));
424  perrors(i,2) = RAD_TO_ASEC*omb.pixel_size_rad*(perrors(i,2));
425  perrors(i,3) = RAD_TO_ASEC*STD_TO_FWHM*omb.pixel_size_rad*(perrors(i,3));
426  perrors(i,4) = RAD_TO_ASEC*STD_TO_FWHM*omb.pixel_size_rad*(perrors(i,4));
427 
428  // if in radec calculate absolute pointing
429  if (telescope.pixel_axes=="radec") {
430  Eigen::VectorXd lat(1), lon(1);
431  lat << params(i,2)*ASEC_TO_RAD;
432  lon << params(i,1)*ASEC_TO_RAD;
433 
434  auto [adec, ara] = engine_utils::tangent_to_abs(lat, lon, omb.wcs.crval[0]*DEG_TO_RAD, omb.wcs.crval[1]*DEG_TO_RAD);
435 
436  params(i,1) = ara(0)*RAD_TO_DEG;
437  params(i,2) = adec(0)*RAD_TO_DEG;
438 
439  perrors(i,1) = perrors(i,1)*ASEC_TO_DEG;
440  perrors(i,2) = perrors(i,2)*ASEC_TO_DEG;
441  }
442  }
443  return 0;
444  });
445 }
