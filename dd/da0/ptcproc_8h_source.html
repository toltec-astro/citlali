134 void PTCProc::subtract_mean(TCData<TCDataKind::PTC, Eigen::MatrixXd> &in) {
135  // cast flags to double and flip 1's and 0's so we can multiply by the data
136  auto f = (in.flags.data.derived().array().cast <double> ().array() - 1).abs();
137  // mean of each detector
138  Eigen::RowVectorXd col_mean = (in.scans.data.derived().array()*f).colwise().sum()/
139  f.colwise().sum();
140 
141  // remove nans from completely flagged detectors
142  Eigen::RowVectorXd dm = (col_mean).array().isNaN().select(0,col_mean);
143 
144  // subtract mean from data and copy into det matrix
145  in.scans.data.noalias() = in.scans.data.derived().rowwise() - dm;
146 
147  // subtract kernel mean
148  if (in.kernel.data.size()!=0) {
149  Eigen::RowVectorXd col_mean = (in.kernel.data.derived().array()*f).colwise().sum()/
150  f.colwise().sum();
151 
152  // remove nans from completely flagged detectors
153  Eigen::RowVectorXd dm = (col_mean).array().isNaN().select(0,col_mean);
154 
155  // subtract mean from data and copy into det matrix
156  in.kernel.data.noalias() = in.kernel.data.derived().rowwise() - dm;
157  }

286 template <typename apt_type, class tel_type, typename Derived>
287 void PTCProc::calc_weights(TCData<TCDataKind::PTC, Eigen::MatrixXd> &in, apt_type &apt, tel_type &telescope,
288  Eigen::DenseBase<Derived> &det_indices) {
289  // number of detectors
290  Eigen::Index n_dets = in.scans.data.cols();
291 
292  // resize weights to number of detectors
293  in.weights.data = Eigen::VectorXd::Zero(n_dets);
294 
295  // approximate weighting
296  if (weighting_type == "approximate") {
297  logger->debug("calculating weights using detector sensitivities");
298  // unit conversion x flux calibration factor x 1/exp(-tau)
299  double conversion_factor;
300 
301  // loop through detectors and calculate weights
302  for (Eigen::Index i=0; i<n_dets; ++i) {
303  // current detector index
304  Eigen::Index det_index = det_indices(i);
305  // if flux calibrated, get flux conversion factor
306  if (in.status.calibrated) {
307  conversion_factor = in.fcf.data(i);
308  }
309  // otherwise fcf is unity
310  else {
311  conversion_factor = 1;
312  }
313  // make sure flux conversion is not zero (otherwise weight=0)
314  if (conversion_factor*apt["sens"](det_index)!=0) {
315  // calculate weights while applying flux calibration
316  in.weights.data(i) = pow(sqrt(telescope.d_fsmp)*apt["sens"](det_index)*conversion_factor,-2.0);
317  }
318  else {
319  in.weights.data(i) = 0;
320  }
321  }
322  }
323  // use full weighting
324  else if (weighting_type == "full"){
325  logger->debug("calculating weights using timestream variance");
326 
327  // loop through detectors
328  for (Eigen::Index i=0; i<n_dets; ++i) {
329  // only calculate weights if detector is unflagged
330  if (apt["flag"](det_indices(i))==0) {
331  // make Eigen::Maps for each detector's scan
332  Eigen::Map<Eigen::Matrix<double, Eigen::Dynamic, 1>> scans(
333  in.scans.data.col(i).data(), in.scans.data.rows());
334  Eigen::Map<Eigen::Matrix<bool, Eigen::Dynamic, 1>> flags(
335  in.flags.data.col(i).data(), in.flags.data.rows());
336 
337  // unflagged detector stddev
338  double det_std_dev = engine_utils::calc_std_dev(scans, flags);
339  // if stddev is not zero
340  if (det_std_dev !=0) {
341  // weight = 1/(stddev)^2
342  in.weights.data(i) = pow(det_std_dev,-2);
343  }
344  // otherwise weight = 0 (not included in maps)
345  else {
346  in.weights.data(i) = 0;
347  }
348  }
349  // otherwise weight = 0 (not included in maps)
350  else {
351  in.weights.data(i) = 0;
352  }
353  }
354  }
355  // constant weighting
356  else if (weighting_type == "const") {
357  for (Eigen::Index i=0; i<n_dets; ++i) {
358  // only calculate weights if detector is unflagged
359  if (apt["flag"](det_indices(i))==0) {
360  in.weights.data(i) = 1;
361  }
362  // otherwise set to zero
363  else {
364  in.weights.data(i) = 0;
365  }
366  }
367  }
368 }

371 auto PTCProc::reset_weights(TCData<TCDataKind::PTC, Eigen::MatrixXd> &in, calib_t &calib,
372  Eigen::DenseBase<Derived> &det_indices) {
373 
374  // only need to run if median weight factor >=1
375  if (med_weight_factor >= 1) {
376  // number of detectors
377  Eigen::Index n_dets = in.scans.data.cols();
378 
379  // get group limits
380  auto grp_limits = get_grouping("array", det_indices, calib, n_dets);
381 
382  // collect detectors that are un-flagged and have non-zero weights
383  for (auto const& [key, val] : grp_limits) {
384  // weights for current group
385  auto grp_weights = in.weights.data(Eigen::seq(std::get<0>(grp_limits[key]),
386  std::get<1>(grp_limits[key])-1));
387  // number of good detectors
388  Eigen::Index n_good_dets = 0;
389  // start index of current group
390  Eigen::Index j = std::get<0>(grp_limits[key]);
391 
392  // loop through detectors in current group
393  for (Eigen::Index m=0; m<grp_weights.size(); m++) {
394  // if detector is good and weight is not zero
395  if (calib.apt["flag"](det_indices(j))==0 && grp_weights(m)>0) {
396  n_good_dets++;
397  }
398  j++;
399  }
400 
401  // to hold good detectors
402  Eigen::VectorXd good_wt;
403 
404  // if good detectors were found
405  if (n_good_dets>0) {
406  good_wt.resize(n_good_dets);
407 
408  // remove flagged dets
409  j = std::get<0>(grp_limits[key]);
410  Eigen::Index k = 0;
411  for (Eigen::Index m=0; m<grp_weights.size(); m++) {
412  if (calib.apt["flag"](det_indices(j))==0 && grp_weights(m)>0) {
413  good_wt(k) = grp_weights(m);
414  k++;
415  }
416  j++;
417  }
418  }
419  // otherwise just use all detectors
420  else {
421  good_wt = grp_weights;
422  }
423 
424  // get median weight
425  auto med_wt = tula::alg::median(good_wt);
426  // store median weights
427  in.median_weights.data.push_back(med_wt);
428 
429  int outliers = 0;
430 
431  // start index of current group
432  j = std::get<0>(grp_limits[key]);
433  // loop through detectors in current group
434  for (Eigen::Index m=0; m<grp_weights.size(); m++) {
435  // if detector weight is med_weight_factor times larger than med_wt
436  if (in.weights.data(j) > med_weight_factor*med_wt) {
437  // reset high weights to median
438  in.weights.data(j) = med_wt;
439  outliers++;
440  }
441  j++;
442  }
443  logger->info("array {} had {} outlier weights", key, outliers);
444  }
445  }
