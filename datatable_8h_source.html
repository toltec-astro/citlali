113  if (v < 0) {

117  }

135  static decltype(auto)
136  parse(IStream &is, std::vector<std::string> *header = nullptr,
137  std::vector<std::vector<std::string>> *meta = nullptr,
138  const std::vector<int> &usecols = {},
139  const std::string &delim = " \t") {
140  SPDLOG_TRACE("parse as ascii, usecols={} delim=\"{}\"", usecols, delim);
141  // is.exceptions(std::ios_base::failbit | std::ios_base::badbit);
142  std::string line;
143  std::string strnum;
144  std::vector<std::vector<Scalar>> data;
145  std::vector<std::vector<std::string>> nondata;
146  // parse line by line
147  while (std::getline(is, line)) {
148  std::vector<Scalar> buf_data;
149  std::vector<std::string> buf_nondata;
150  bool is_data = true;
151  for (auto i = line.begin(); i != line.end(); ++i) {
152  if (!isascii(*i)) {
153  throw ParseError("not an ascii file");
154  }
155  // If i is not a delim, then append it to strnum
156  if (delim.find(*i) == std::string::npos) {
157  strnum += *i;
158  // continue to next char unless this is the last one
159  if (i + 1 != line.end())
160  continue;
161  }
162  // if strnum is still empty, it means the previous char is also
163  // a delim (several delims appear together). Ignore this char.
164  if (strnum.empty())
165  continue;
166  // If we reach here, we got something.
167  // store a backup in the header in case it is not a number
168  buf_nondata.push_back(strnum);
169  // try convert it to number.
170  if (is_data) {
171  std::istringstream ss(strnum);
172  Scalar number;
173  ss >> number;
174  if (ss.fail()) {
175  // not a number, treat as header
176  is_data = false;
177  }
178  if (is_data) {
179  buf_data.push_back(number);
180  }
181  }
182  strnum.clear();
183  }
184  if (is_data) {
185  data.emplace_back(buf_data);
186  } else {
187  nondata.emplace_back(buf_nondata);
188  }
189  }
190  // handle nondata
191  if (header != nullptr) {
192  *header = nondata[0];
193  if (header[0][0] == "#") {
194  header->erase(header->begin());
195  }
196  }
197  if (meta != nullptr) {
198  *meta = nondata;
199  }
200  // convert to Eigen matrix
201  auto nrows = static_cast<Index>(data.size());
202  auto ncols = static_cast<Index>(data[0].size());
203  auto ncols_use = ncols;
204  SPDLOG_TRACE("shape of table ({}, {})", nrows, ncols);
205  // get the usecols
206  if (!usecols.empty()) {
207  for (auto i : usecols) {
208  if ((i < -ncols) || (i >= ncols))
209  throw ParseError(fmt::format(
210  "invalid column index {} for table of ncols={}", i,
211  ncols));
212  }
213  ncols_use = meta::size_cast<Index>(usecols.size());
214  SPDLOG_TRACE("using {} cols out of {}", ncols_use, ncols);
215  }
216  using Eigen::Dynamic;
217  using Eigen::Map;
218  using Eigen::Matrix;
219  Matrix<Scalar, Dynamic, Dynamic> ret(nrows, ncols_use);
220  for (Index i = 0; i < nrows; ++i) {
221  if (usecols.empty()) {
222  ret.row(i) =
223  Map<Matrix<Scalar, Dynamic, 1>>(&data[i][0], ncols_use);
224  } else {
225  for (std::size_t j = 0; j < usecols.size(); ++j) {
226  auto v = usecols[j];
227  if (v < 0)
228  v += ncols;
229  // SPDLOG_TRACE("get table data {} {} to return {} {}", i,
230  // v, i, j);
231  ret(i, j) = data[i][v];
232  }
233  }
234  data[i].clear();
235  }
236  return ret;
237  }

260  Index ncols = Eigen::Dynamic,

263  using Eigen::Matrix;
264  using Eigen::RowMajor;
265  // throw everything

267  std::ios_base::eofbit);
268  // get file size
269  is.seekg(0, std::ios_base::end);
270  auto filesize = is.tellg();
271  auto size = filesize / sizeof(Scalar);
272  SPDLOG_TRACE("memdump size={} nelem={}", filesize, size);
273  // validate based on nrows and ncols
274  if (ncols == Dynamic) {
275  ncols = 1;
276  } else if (size % ncols != 0) {
277  throw ParseError(fmt::format(
278  "memdump size {} inconsistent with ncols={}", size, ncols));
279  }
280  if (nrows == Dynamic) {
281  nrows = size / ncols;
282  } else if (nrows * ncols != size) {
283  throw ParseError(fmt::format(
284  "memdump size {} inconsistent with nrows={} ncols={}", size,
285  nrows, ncols));
286  }
287  SPDLOG_TRACE("memdump shape ({}, {})", nrows, ncols);
288  // this is colmajor
289  Matrix<Scalar, Dynamic, Dynamic> ret{nrows, ncols};
290  auto get = [&](auto *data) {
291  is.seekg(0, std::ios_base::beg);
292  is.read(reinterpret_cast<char *>(data), filesize);
293  };
294  if (order & Eigen::RowMajor) {
295  Matrix<Scalar, Dynamic, Dynamic, RowMajor> tmp{nrows, ncols};
296  get(tmp.data());
297  ret = tmp;
298  } else {
299  get(ret.data());
300  }
301  SPDLOG_TRACE("memdump data{:r10c10}", ret);
