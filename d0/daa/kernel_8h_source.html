124  double off_lon = 0.0;

128  double sigma_lon = sigma_rad;

138  // distance to source to truncate it
139  auto dist = ((lat.array()).pow(2) + (lon.array()).pow(2)).sqrt();
140 
141  // calculate stddev from apt table if config stddev <=0
142  if (sigma_rad <= 0) {
143  sigma_lat = FWHM_TO_STD * ASEC_TO_RAD * apt["b_fwhm"](det_index);
144  sigma_lon = FWHM_TO_STD * ASEC_TO_RAD * apt["a_fwhm"](det_index);
145  }
146 
147  // rotation angle
148  double rot_ang = apt["angle"](det_index);
149 
150  auto cost2 = cos(rot_ang) * cos(rot_ang);
151  auto sint2 = sin(rot_ang) * sin(rot_ang);
152  auto sin2t = sin(2. * rot_ang);
153  auto xstd2 = sigma_lon * sigma_lon;
154  auto ystd2 = sigma_lat * sigma_lat;
155  auto a = - 0.5 * ((cost2 / xstd2) + (sint2 / ystd2));
156  auto b = - 0.5 * ((sin2t / xstd2) - (sin2t / ystd2));
157  auto c = - 0.5 * ((sint2 / xstd2) + (cost2 / ystd2));
158 
159  double sigma_limit_det = sigma_limit * (sigma_lat + sigma_lon)/2;
160 
161  // make elliptical gaussian
162  for (Eigen::Index j=0; j<n_pts; ++j) {
163  // truncate within radius
164  if (dist(j) <= sigma_limit_det) {
165  in.kernel.data(j,i) = amp*exp(pow(lon(j) - off_lon, 2) * a +
166  (lon(j) - off_lon) * (lat(j) - off_lat) * b +
167  pow(lat(j) - off_lat, 2) * c);
168  }
169  else {
170  in.kernel.data(j,i) = 0;
171  }
172  }
