173 auto mapFitter::fit_to_gaussian(Eigen::DenseBase<Derived> &signal, Eigen::DenseBase<Derived> &weight,
174  double init_fwhm, double init_row, double init_col) {
175 
176  // initial parameters and limits
177  Eigen::VectorXd init_params(n_params);
178  Eigen::MatrixXd limits(n_params,2);
179 
180  // intiial position and flux
181  double init_flux = 0;
182 
183  // initial gaussian standard deviation
184  double init_sigma = init_fwhm*FWHM_TO_STD;
185 
186  // if no initial position is input find peak
187  if (init_row<0 && init_col<0) {
188  // center positions
189  double center_row = (signal.rows() - 1)/2;
190  double center_col = (signal.cols() - 1)/2;
191 
192  // signal-to-noise map
193  auto sig2noise = signal.derived().array()*sqrt(weight.derived().array());
194 
195  Eigen::Index ir, ic;
196 
197  // find peak in the entire map
198  if (fitting_region_pix <= 0) {
199  sig2noise.maxCoeff(&ir, &ic);
200  init_flux = signal(ir,ic);
201  }
202  // find peak within inner radius
203  else {
204  for (Eigen::Index i=0; i<sig2noise.rows(); ++i) {
205  for (Eigen::Index j=0; j<sig2noise.cols(); ++j) {
206  auto dist = sqrt(pow(i - center_row,2) + pow(j - center_col,2));
207  if (dist < fitting_region_pix) {
208  if (sig2noise(i,j) > init_flux) {
209  init_flux = sig2noise(i,j);
210  ir = i;
211  ic = j;
212  }
213  }
214  }
215  }
216  // initial guess for flux
217  init_flux = signal(ir, ic);
218  }
219 
220  init_row = ir;
221  init_col = ic;
222  }
223  // otherwise use the input initial position
224  else {
225  init_flux = signal(static_cast<int>(init_row), static_cast<int>(init_col));
226  }
227 
228  // initial parameter guesses (order of positions is x,y = col,row)
229  init_params << init_flux, init_col, init_row, init_sigma, init_sigma, 0;
230 
231  // limits of bounding box
232  double lower_row, lower_col, upper_row, upper_col;
233 
234  // ignore bounding box if less than/equal to zero
235  if (bounding_box_pix <= 0) {
236  lower_row = 0;
237  upper_row = 0;
238 
239  upper_row = signal.rows() - 1;
240  upper_col = signal.cols() - 1;
241  }
242  // determine bounding box size
243  else {
244  // ensure lower limits of bounding box are not less than zero
245  lower_row = std::max(init_row - bounding_box_pix, 0.0);
246  lower_col = std::max(init_col - bounding_box_pix, 0.0);
247 
248  // ensure upper limits of bounding box are not bigger than the map
249  upper_row = std::min(init_row + bounding_box_pix, static_cast<double>(signal.rows()) - 1);
250  upper_col = std::min(init_col + bounding_box_pix, static_cast<double>(signal.cols()) - 1);
251  }
252 
253  // size of bounding box region
254  double n_rows = upper_row - lower_row + 1;
255  double n_cols = upper_col - lower_col + 1;
256 
257  // set lower limits of fitting parameters
258  limits.col(0) << flux_low*init_flux, lower_col, lower_row, fwhm_low*init_sigma,
259  fwhm_low*init_sigma, angle_low;
260 
261  // set upper limits of fitting parameters
262  limits.col(1) << flux_high*init_flux, upper_col, upper_row, fwhm_high*init_sigma,
263  fwhm_high*init_sigma, angle_high;
264 
265  Eigen::VectorXd x, y;
266 
267  // axes coordinate vectors for meshgrid
268  x = Eigen::VectorXd::LinSpaced(n_cols, lower_col, upper_col);
269  y = Eigen::VectorXd::LinSpaced(n_rows, lower_row, upper_row);
270 
271  // create gaussian 2d model
272  auto g = create_model<Gaussian2D>(init_params);
273  // get meshgrid
274  auto xy = g.meshgrid(x, y);
275 
276  // get map stddev
277  double map_sigma = engine_utils::calc_std_dev(signal);
278 
279  // standard deviation of signal map
280  Eigen::MatrixXd sigma(weight.rows(), weight.cols());
281 
282  // loop through pixels
283  for (Eigen::Index i=0; i<weight.rows(); ++i) {
284  for (Eigen::Index j=0; j<weight.cols(); ++j) {
285  if (weight(i,j)!=0) {
286  // use map sigma for beammaps
287  if constexpr (fit_mode == FitMode::beammap) {
288  sigma(i,j) = map_sigma;
289  }
290  // use weights for pointing
291  else if constexpr (fit_mode == FitMode::pointing) {
292  sigma(i,j) = 1./sqrt(weight(i,j));
293  }
294  }
295  else {
296  sigma(i,j) = 0;
297  }
298  }
299  }
300 
301  // copy data and sigma within bounding box region
302  Eigen::MatrixXd _signal = signal.block(lower_row, lower_col, n_rows, n_cols);
303  Eigen::MatrixXd _sigma = sigma.block(lower_row, lower_col, n_rows, n_cols);
304 
305  // do the fit and return
306  return ceres_fit(g, init_params, xy, _signal, _sigma, limits);
307 }
