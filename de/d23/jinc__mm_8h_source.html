76 auto JincMapmaker::jinc_func(double r, double a, double b, double c, double r_max, double l_d) {
77  if (r!=0) {
78  // unitless radius
79  r = r/l_d;
80  // first jinc function
81  auto jinc_1 = 2.*boost::math::cyl_bessel_j(1,2.*pi*r/a)/(2.*pi*r/a);
82  // exponential
83  auto exp_func = exp(-pow(2.*r/b,c));
84  // second jinc function
85  auto jinc_2 = 2.*boost::math::cyl_bessel_j(1,3.831706*r/r_max)/(3.831706*r/r_max);
86  // jinc1 x exp x jinc2
87  return jinc_1*exp_func*jinc_2;
88  }
89  else {
90  return 1.0;
91  }
92 }

94 void JincMapmaker::allocate_jinc_matrix(double pixel_size_rad) {
95  l_d[0] = (1.1/1000)/50;
96  l_d[1] = (1.4/1000)/50;
97  l_d[2] = (2.0/1000)/50;
98 
99  // loop through lambda/diameters
100  for (const auto &ld: l_d) {
101  // get shape params
102  auto a = shape_params[ld.first](0);
103  auto b = shape_params[ld.first](1);
104  auto c = shape_params[ld.first](2);
105 
106  // maximum radius in pixels
107  int r_max_pix = std::floor(r_max*ld.second/pixel_size_rad);
108 
109  // pixel centers within max radius
110  Eigen::VectorXd pixels = Eigen::VectorXd::LinSpaced(2*r_max_pix + 1,-r_max_pix, r_max_pix);
111 
112  // allocate jinc weights
113  jinc_weights_mat[ld.first].setZero(2*r_max_pix + 1,2*r_max_pix + 1);
114 
115  // loop through matrix rows
116  for (Eigen::Index i=0; i<pixels.size(); ++i) {
117  // loop through matrix cols
118  for (Eigen::Index j=0; j<pixels.size(); ++j) {
119  // radius of current pixel in radians
120  double r = pixel_size_rad*sqrt(pow(pixels(i),2) + pow(pixels(j),2));
121  // calculate jinc weight at pixel
122  jinc_weights_mat[ld.first](i,j) = jinc_func(r,a,b,c,r_max,ld.second);
123  }
124  }
125  }

128 void JincMapmaker::calculate_jinc_splines() {
129  l_d[0] = (1.1/1000)/50;
130  l_d[1] = (1.4/1000)/50;
131  l_d[2] = (2.0/1000)/50;
132 
133  // loop through lambda/diameters
134  for (const auto &ld: l_d) {
135  // get shape params
136  auto a = shape_params[ld.first](0);
137  auto b = shape_params[ld.first](1);
138  auto c = shape_params[ld.first](2);
139 
140  // radius vector in radians
141  auto radius = Eigen::VectorXd::LinSpaced(n_pts_splines, 0, r_max*ld.second);
142  // jinc weights on dense vector
143  Eigen::VectorXd jinc_weights(radius.size());
144 
145  Eigen::Index j = 0;
146 
147  for (const auto &r: radius) {
148  // calculate jinc weights
149  jinc_weights(j) = jinc_func(r,a,b,c,r_max,ld.second);
150  ++j;
151  }
152  // create spline class
153  engine_utils::SplineFunction s;
154  // spline interpolate
155  s.interpolate(radius, jinc_weights);
156  // store jinc spline
157  jinc_splines[ld.first] = s;
