348 
349  // check tau (may be unnecessary)
350  if (!telescope.sim_obs) {
351  Eigen::VectorXd tau_el(1);
352  // get mean elevation
353  tau_el << telescope.tel_data["TelElAct"].mean();
354  // get tau at mean elevation for each band
355  auto tau_freq = rtcproc.calibration.calc_tau(tau_el, telescope.tau_225_GHz);
356  // loop through and make sure average tau is not negative (implies wrong model)
357  for (auto const& [key, val] : tau_freq) {
358  if (val[0] < 0) {
359  logger->error("calculated mean {} tau {} < 0",toltec_io.array_name_map[key], val[0]);
360  std::exit(EXIT_FAILURE);
361  }
362  }
363  }
364  }
365 
366  else {
367  rtcproc.calibration.extinction_model = "N/A";
368  }
369 
370  // make sure there are matched fg's in apt if reducing in polarized mode
371  if (rtcproc.run_polarization) {
372  if ((calib.apt["fg"].array()==-1).all()) {
373  logger->error("no matched freq groups. cannot run in polarized mode");
374  std::exit(EXIT_FAILURE);
375  }
376  }

380  rtcproc.kernel.setup(n_maps);

382

393  rtcproc.filter.w0s.push_back(24.46);

413  create_obs_map_files();
414  }
415  }

417  if (run_tod_output) {

427  // make ptc tod output file

430  }

433  // don't calculate any eigenvalues

453  }

455  else {

458  }
459  else {

461  }
462  }
463

471  }
472  // clear stored eigenvalues
473  diagnostics.evals.clear();
474 }

1073  // conversion to Jy/pixel
1074  add_netcdf_var(fo, "to_Jy_pixel_"+name, mJy_beam_to_Jy_px);

1077  // conversion to mJy/beam
