33  auto get_stddev_index(const Eigen::DenseBase<Derived> &evals) {
34  // copy eigenvalues
35  Eigen::VectorXd ev = evals.derived().array().abs().log10();
36 
37  auto n_dets = evals.size();
38  // mean of eigenvalues
39  auto m_ev = ev.mean();
40  // standard deviation of eigenvalues
41  auto stddev = engine_utils::calc_std_dev(ev);
42 
43  bool keep_going = true;
44  int n_keep_last = n_dets;
45 
46  // vector of eigenvalues below stddev cut
47  Eigen::Matrix<bool,Eigen::Dynamic,1> good(n_dets);
48  good.setOnes(n_dets);
49 
50  int iterator = 0;
51  while (keep_going) {
52  // count up number of eigenvalues that pass the cut
53  int count = 0;
54  for (Eigen::Index i=0; i<n_dets; i++) {
55  if (good(i)) {
56  if (abs(ev(i) - m_ev) > abs(stddev_limit*stddev)) {
57  good(i) = false;
58  }
59  else {
60  count++;
61  }
62  }
63  }
64 
65  if (count >= n_keep_last) {
66  keep_going = false;
67  }
68  else {
69  // get mean for good eigen values
70  m_ev = 0.;
71  for (Eigen::Index i=0; i<n_dets; i++) {
72  if (good(i)) {
73  m_ev += ev(i);
74  }
75  }
76  // get stddev for good eigen values
77  m_ev /= count;
78  stddev = 0.;
79  for (Eigen::Index i=0; i<n_dets; i++) {
80  if (good(i)) {
81  stddev += (ev(i) - m_ev)*(ev(i) - m_ev);
82  }
83  }
84  stddev = stddev/(count-1.);
85  stddev = sqrt(stddev);
86  n_keep_last = count;
87  }
88  iterator++;
89  }
90 
91  // stddev limit
92  double limit = pow(10.,m_ev + stddev_limit*stddev);
93  // index where limit occurs
94  Eigen::Index limit_index = 0;
95 
96  // find index
97  for (Eigen::Index i=0; i<n_dets; i++) {
98  if (evals(i) <= limit){
99  limit_index = i;
100  break;
101  }
102  }
103 
104  return limit_index;
105  }
